274 {
275  auto F = MemoryBuffer::getFile(Filename);

384 bool
385 PolicyFile::FieldReadHook(const llvm::StructType& T, StringRef Field) const
386 {
387  if (not T.getName().startswith("struct.")) {
388  assert(T.getName().startswith("union."));
389  return false;
390  }
391 
392  StringRef Name = T.getName().substr(7);
393

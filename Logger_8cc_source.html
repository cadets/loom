169 LibxoLogger::CreateFormatString(IRBuilder<>& Builder, StringRef Prefix,
170  ArrayRef<Parameter> Params, StringRef Suffix,
171  bool SuppressUniqueness) {
172 
173  std::stringstream FormatString;
174 
175  FormatString << Prefix.str();
176 
177  for (auto& P : Params) {
178  const string Name = P.first;
179  Type *T = P.second;
180 
181  // xo can humanize values (e.g., 41025981 -> 41M), but we don't want to
182  // do this with pointer values (e.g., 0x7fff01... -> 128T) or
183  // floating-point numbers (13.415235 -> 13).
184  const bool Humanize = T->isIntegerTy();
185 
186  FormatString
187  << "{P: }" // padding
188  << "{"
189  << (Humanize ? "h" : "")
190  << ":"
191  << (SuppressUniqueness ? "" : Name)
192  << "/"
193  ;
194 
195  if (T->isIntegerTy(32)) {
196  FormatString << "%d";
197 
198  } else if (T->isFloatTy() || T->isDoubleTy()) {
199  FormatString << "%f";
200 
201  } else if (T->isIntegerTy(8)) {
202  FormatString << "%c";
203 
204  } else if (T->isPointerTy()
205  and T->getPointerElementType()->isIntegerTy(8)) {
206  FormatString << "\"%s\"";
207 
208  } else if (T->isPointerTy()) {
209  FormatString << "%p";
210 
211  }
212 
213  FormatString << "}"; // close the format string
214  }
215 
216  FormatString << Suffix.str();
217 
218  const string Str = FormatString.str();
219 
220  auto i = FormatStrings.find(Str);
221  if (i != FormatStrings.end()) {
222  return i->second;
223  }
224 
225  Value *Ptr = Builder.CreateGlobalStringPtr(Str);
226  auto *GV = dyn_cast<GlobalVariable>(Ptr->stripInBoundsConstantOffsets());
227  GV->setUnnamedAddr(GlobalValue::UnnamedAddr::Global);
228  GV->setAlignment(1);
229 
230  FormatStrings[Str] = Ptr;
231 
232  return Ptr;
233 }

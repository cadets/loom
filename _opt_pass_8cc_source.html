70 bool OptPass::runOnModule(Module &Mod)
71 {
72  if (std::error_code err = PolFile.getError()) {
73  errs()
74  << "Error opening LOOM policy file '" << PolicyFilename
75  << "': " << err.message() << "\n";
76  return false;
77  }
78 
79  DebugInfo Debug(Mod);
80  if (not Debug.ModuleHasFullDebugInfo()) {
81  errs()
82  << "Warning: module missing metadata, instrumentation may be incomplete\n"
83  ;
84  }
85 
86  assert(*PolFile);
87  Policy& P = **PolFile;
88 
89  Instrumenter::NameFn Name =
90  [&P](const std::vector<std::string>& Components)
91  {
92  return P.InstrName(Components);
93  };
94 
95  auto S = InstrStrategy::Create(P.Strategy(), P.UseBlockStructure());
96 
97  for (auto& L : P.Loggers(Mod)) {
98  S->AddLogger(std::move(L));
99  }
100 
101  unique_ptr<Instrumenter> Instr(Instrumenter::Create(Mod, Name, std::move(S)));
102 
103  //
104  // In order to keep from invalidating iterators or instrumenting our
105  // instrumentation, we need to decide on the instruction-oriented
106  // instrumentation points like calls before we actually instrument them.
107  //
108  std::vector<Instruction*> AllInstructions;
109 
110  std::unordered_map<Function*, Policy::Directions> Functions;
111  std::unordered_map<CallInst*, Policy::Directions> Calls;
112 
113  typedef std::pair<GetElementPtrInst*, std::string> FieldGEP;
114  std::unordered_map<LoadInst*, FieldGEP> FieldReads;
115  std::unordered_map<StoreInst*, FieldGEP> FieldWrites;
116 
117  for (auto& Fn : Mod) {
118  // Do we need to instrument this function?
119  Policy::Directions Directions = P.FnHooks(Fn);
120  if (not Directions.empty()) {
121  Functions.emplace(&Fn, Directions);
122  }
123 
124  for (auto& Inst : instructions(Fn)) {
125  if (P.InstrumentAll()) {
126  AllInstructions.push_back(&Inst);
127  }
128 
129  if (auto *GEP = dyn_cast<GetElementPtrInst>(&Inst)) {
130  if (auto *ST = dyn_cast<StructType>(GEP->getSourceElementType())) {
131  // A GEP used for structure field lookup should have indices
132  // 0 and i, where i is the field number (not byte index).
133  if (GEP->getNumIndices() != 2 or !GEP->hasAllConstantIndices())
134  continue;
135 
136  if (not P.StructTypeMatters(*ST))
137  continue;
138 
139  std::string FieldName = Debug.FieldName(GEP);
140  assert(not FieldName.empty());
141 
142  const bool HookReads = P.FieldReadHook(*ST, FieldName);
143  const bool HookWrites = P.FieldWriteHook(*ST, FieldName);
144 
145  if (not HookReads and not HookWrites) {
146  continue;
147  }
148 
149  for (auto& Use : GEP->uses()) {
150  User *U = Use.getUser();
151 
152  if (HookReads) {
153  if (auto *Load = dyn_cast<LoadInst>(U)) {
154  FieldReads[Load] = { GEP, FieldName };
155  }
156  }
157 
158  if (HookWrites) {
159  if (auto *Store = dyn_cast<StoreInst>(U)) {
160  FieldWrites[Store] = { GEP, FieldName };
161  }
162  }
163  }
164  }
165  }
166 
167  // Is this a call to instrument?
168  if (CallInst* Call = dyn_cast<CallInst>(&Inst)) {
169  Function *Target = Call->getCalledFunction();
170  if (not Target)
171  continue; // TODO: support indirect targets
172 
173  Policy::Directions Directions = P.CallHooks(*Target);
174  if (not Directions.empty())
175  Calls.emplace(Call, Directions);
176  }
177  }
178  }
179 
180  //
181  // Now we actually perform the instrumentation:
182  //
183  bool ModifiedIR = false;
184 
185  for (auto *I : AllInstructions) {
186  Instr->Instrument(I);
187  }
188 
189  for (auto& i : Functions) {
190  ModifiedIR |= Instr->Instrument(*i.first, i.second);
191  }
192 
193  for (auto& i : Calls) {
194  ModifiedIR |= Instr->Instrument(i.first, i.second);
195  }
196 
197  for (auto& i : FieldReads) {
198  LoadInst *Load = i.first;
199  GetElementPtrInst *GEP = i.second.first;
200  StringRef FieldName = i.second.second;
201 
202  ModifiedIR |= Instr->Instrument(GEP, Load, FieldName);
203  }
204 
205  for (auto& i : FieldWrites) {
206  StoreInst *Store = i.first;
207  GetElementPtrInst *GEP = i.second.first;
208  StringRef FieldName = i.second.second;
209 
210  ModifiedIR |= Instr->Instrument(GEP, Store, FieldName);
211  }
212 
213  return ModifiedIR;
214 }

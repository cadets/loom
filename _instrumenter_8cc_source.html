59 bool Instrumenter::Instrument(llvm::Instruction *I)
60 {
61  // If this instruction terminates a block, we need to treat it a bit
62  // differently, placing instrumentation before it rather than after.
63  const bool Terminator = I->isTerminator();
64 
65  // If this instruction is a terminator *or* is of void type, it doesn't
66  // have a value for us to log.
67  const bool Void = Terminator or I->getType()->isVoidTy();
68 
69  auto *OpcodeTy = IntegerType::get(Mod.getContext(), 32);
70  unsigned Opcode = I->getOpcode();
71 
72  ParamVec ValueDescriptions = { { "opcode", OpcodeTy } };
73  vector<Value*> Values = { ConstantInt::get(OpcodeTy, Opcode) };
74 
75  if (not Void)
76  {
77  ValueDescriptions.emplace_back(I->getName(), I->getType());
78  Values.push_back(I);
79  }
80 
81  for (Use& U : I->operands()) {
82  // If this is a phi node, ignore the operands (values that *could* exist)
83  // and stick to the phi value (that value that *does* exist).
84  if (isa<PHINode>(I)) {
85  break;
86  }
87 
88  Value *V = U.get();
89 
90  // Don't try to pass a BasicBlock around by value (e.g., in a BranchInst).
91  if (isa<BasicBlock>(V)) {
92  continue;
93  }
94 
95  // Don't use the address of an LLVM intrinsic: report its name instead.
96  if (Function *F = dyn_cast<Function>(V)) {
97  if (F->hasLLVMReservedName()) {
98  V = IRBuilder<>(I).CreateGlobalStringPtr(F->getName(), "fn_name");
99  }
100  }
101 
102  Type *T = V->getType();
103  if (T->isMetadataTy()) {
104  // Ignore metadata-consuming instructions such as calls to
105  // @llvm.dbg.declare().
106  return false;
107  }
108 
109  ValueDescriptions.emplace_back(V->getName(), T);
110  Values.push_back(V);
111  }
112 
113  // We don't handle varargs generically: that needs to be done by
114  // function- or call-specific instrumentation.
115  constexpr bool Varargs = false;
116 
117  // Instrument all non-terminators after the instructor, so that we can
118  // capture the instruction's value (if non-void).
119  const bool AfterInst = not Terminator;
120 
121  // Every instrumentation point needs a unique name. Eventually we should do
122  // something clever with debug information, but for now we'll use a more...
123  // simplistic approach.
124  std::ostringstream NameBuilder;
125  NameBuilder << "instrumentation:instruction:";
126  NameBuilder << static_cast<const void*>(I);
127  const string Name = NameBuilder.str();
128 
129  Strategy->Instrument(I, Name, Name, ValueDescriptions, Values,
130  Varargs, AfterInst, true);
131 
132  return true;
133 }

147 bool Instrumenter::Instrument(llvm::CallInst *Call, Policy::Direction Dir)
148 {
149  Function* Target = Call->getCalledFunction();
150  assert(Target); // TODO: support indirect targets, too
151 
152  // Get some relevant details about the call and the target function.
153  const string TargetName = Target->getName();
154  const bool VarArgs = Target->isVarArg();
155  Type *CallType = Call->getType();
156  const bool voidFunction = CallType->isVoidTy();
157  const bool Return = (Dir == Policy::Direction::Out);
158 
159  const string Description = Return ? "return" : "call";
160  const string FormatStringPrefix = Description + " " + TargetName + ":";
161  const string InstrName = Name({ Description, TargetName });
162 
163  // Start by copying static and dynamic value details from the target function.
164  ParamVec Parameters = GetParameters(Target);
165  vector<Value*> Arguments(Call->getNumArgOperands());
166  std::copy(Call->arg_begin(), Call->arg_end(), Arguments.begin());
167 
168  // The return value, if present, comes first in the instrumentation.
169  if (Return and not voidFunction) {
170  Parameters.emplace(Parameters.begin(), "retval", Call->getType());
171  Arguments.emplace(Arguments.begin(), Call);
172  }
173 
174  bool InstrAfterCall = Return;
175  Strategy->Instrument(Call, InstrName, FormatStringPrefix,
176  Parameters, Arguments, VarArgs, InstrAfterCall);
177 
178  return true;
179 }

194 bool
195 Instrumenter::Instrument(Function& Fn, Policy::Direction Dir) {
196  const bool Return = (Dir == Policy::Direction::Out);
197  const string Description = Return ? "leave" : "enter";
198  StringRef FnName = Fn.getName();
199 
200  assert(isa<PointerType>(Fn.getType()));
201  FunctionType *FnType = dyn_cast<FunctionType>(Fn.getType()->getElementType());
202  assert(FnType);
203  const bool voidFunction = FnType->getReturnType()->isVoidTy();
204 
205  vector<Value*> Arguments;
206  vector<Parameter> InstrParameters;
207 
208  if (Return and not voidFunction) {
209  Arguments.push_back(nullptr);
210  InstrParameters.emplace_back("retval", FnType->getReturnType());
211  }
212 
213  for (auto& Arg : Fn.getArgumentList()) {
214  Arguments.push_back(&Arg);
215  InstrParameters.emplace_back(Arg.getName(), Arg.getType());
216  }
217 
218  const string InstrName = Name({ Description, FnName });
219  string FormatStringPrefix = (Description + " " + FnName + ":").str();
220 
221  if (Return) {
222  // Instrument all returns from the function:
223  SmallVector<ReturnInst*, 4> Returns;
224 
225  for (auto& Block : Fn) {
226  TerminatorInst *Terminator = Block.getTerminator();
227  if (auto *Ret = dyn_cast<ReturnInst>(Terminator)) {
228  Returns.push_back(Ret);
229  }
230  }
231 
232  for (ReturnInst *Ret : Returns) {
233  // This is a return: pass the return value and then the function's
234  // parameters to the instrumentation.
235  if (not voidFunction) {
236  Arguments[0] = Ret->getReturnValue();
237  }
238 
239  Strategy->Instrument(Ret, InstrName, FormatStringPrefix,
240  InstrParameters, Arguments);
241  }
242 
243  } else {
244  // Instrument the function's preamble:
245  assert(not Fn.getBasicBlockList().empty());
246  BasicBlock& Entry = Fn.getBasicBlockList().front();
247 
248  Strategy->Instrument(&Entry.front(), InstrName, FormatStringPrefix,
249  InstrParameters, Arguments);
250  }
251 
252  return false;
253 }

318 uint32_t Instrumenter::FieldNumber(GetElementPtrInst *GEP) {
319  // A field access getelementptr should have two indices:
320  // a zero and the target field number (a constant integer).
321  Use *IdxZero = GEP->idx_begin();
322  Use *FieldIdx = IdxZero + 1;
323  assert(GEP->getNumIndices() == 2);
324  assert(dyn_cast<ConstantInt>(IdxZero->get())->isZero());
325  assert(isa<ConstantInt>(FieldIdx->get()));
326 
327  ConstantInt *FieldNum = dyn_cast<ConstantInt>(FieldIdx->get());
328  return FieldNum->getZExtValue();

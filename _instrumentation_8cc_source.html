78 IRBuilder<> Instrumentation::AddAction(StringRef Name)
79 {
80  assert(PreambleEnd && End);
81 
82  if (not Preamble) {
83  // If we don't have a preamble block, we are doing direct inline
84  // instrumentation and we should not have an end block either.
85  // Return an IRBuilder positioned just before the end instruction.
86  assert(not EndBlock);
87  return IRBuilder<>(End);
88  }
89 
90  assert(EndBlock);
91 
92  // Get the last block in the instrumentation chain before "end"
93  // and unhook it from "end": we need to insert the new block in between.
94  BasicBlock *Predecessor = EndBlock->getSinglePredecessor();
95  assert(Predecessor);
96 
97  EndBlock->removePredecessor(Predecessor);
98  Predecessor->getTerminator()->eraseFromParent();
99 
100  // Create the new instrumentation block and insert it between the
101  // old last-but-one block and the "end" block.
102  Function *Fn = Predecessor->getParent();
103  LLVMContext& Ctx = Fn->getContext();
104  BasicBlock *BB = BasicBlock::Create(Ctx, Name, Fn, EndBlock);
105 
106  IRBuilder<>(Predecessor).CreateBr(BB);
107  BranchInst *Terminator = IRBuilder<>(BB).CreateBr(EndBlock);
108 
109  // Return an IRBuilder positioned immediately before the final branch
110  // to the "end" block.
111  return IRBuilder<>(Terminator);
112 }

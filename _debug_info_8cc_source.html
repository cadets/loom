44 DebugInfo::DebugInfo(llvm::Module& M)
45  : Mod(M), DbgDeclare(Mod.getFunction("llvm.dbg.declare"))
46 {
47  Mod.getMDKindNames(MetadataKinds);

60 bool DebugInfo::ModuleHasFullDebugInfo() const
61 {
62  if (not DbgDeclare)

64

66  if (not ModuleMetadata)

70  constexpr auto FullDebug = DICompileUnit::DebugEmissionKind::FullDebug;
71 
72  return DI->getEmissionKind() == FullDebug and not DI->isOptimized();
73  }
74 
75  return false;

79 std::string DebugInfo::FieldName(GetElementPtrInst *GEP)
80 {
81  // Trace back to a variable with debug metadata.
82  SmallVector<size_t, 4> GEPOffsets;
83  const DIVariable *Var = Trace(GEP, GEPOffsets);
84  if (Var == nullptr) {
85  errs() << "WARNING: unable to trace GEP ";
86  GEP->print(errs(), true);
87  errs() << " back to a variable\n";
88 
89  return "";
90  }
91 
92  const DIType *T = dyn_cast<DIType>(Var->getType());
93  assert(T && "DIVariable::getType() should return a DIType");
94 
95  do {
96  switch (T->getTag()) {
97  case dwarf::DW_TAG_array_type: {
98  auto *CT = dyn_cast<DICompositeType>(T);
99  T = dyn_cast<DIType>(CT->getBaseType());
100  break;
101  }
102 
103  case dwarf::DW_TAG_pointer_type: {
104  auto *DT = dyn_cast<DIDerivedType>(T);
105  T = dyn_cast<DIType>(DT->getBaseType());
106  break;
107  }
108 
109  case dwarf::DW_TAG_structure_type: {
110  assert(not GEPOffsets.empty());
111 
112  size_t Index = GEPOffsets.back();
113  GEPOffsets.pop_back();
114 
115  auto *CT = dyn_cast<DICompositeType>(T);
116  auto *Member = dyn_cast<DIDerivedType>(CT->getElements()[Index]);
117 
118  if (GEPOffsets.empty()) {
119  return Member->getName();
120  } else {
121  T = dyn_cast<DIType>(Member->getBaseType());
122  }
123 
124  break;
125  }
126 
127  default:
128  errs() << "ERROR: unhandled debug info tag: " << T->getTag() << "\n";
129  return "";
130  }
131 
132  } while (T);
133 
134  return "";
135 }

138 const DIVariable*
139 DebugInfo::Trace(GetElementPtrInst *GEP, SmallVectorImpl<size_t> &Offsets)
140 {
141  // Walk backwards from GEP to source until we find a variable with
142  // debug metadata (or die trying, having reached the end of the GEP chain).
143  do
144  {
145  // If GEP'ing the address of a structure field, track which field
146  // index we are working with.
147  if (isa<StructType>(GEP->getSourceElementType())
148  and GEP->getNumIndices() == 2) {
149 
150  Value *Index = GEP->idx_begin()[1];
151 
152  auto *ConstIdx = dyn_cast<ConstantInt>(Index);
153  assert(ConstIdx && "struct field indexing must be static");
154  Offsets.push_back(ConstIdx->getLimitedValue());
155  }
156 
157  Value *Ptr = GEP->getPointerOperand()->stripPointerCasts();
158  if (auto *Var = Get<DIVariable>(Ptr)) {
159  return Var;
160  }
161 
162  if (auto *Load = dyn_cast<LoadInst>(Ptr)) {
163  while (Load != nullptr) {
164  Ptr = Load->getPointerOperand();
165  Load = dyn_cast<LoadInst>(Ptr);
166  }
167  }
168 
169  GEP = dyn_cast<GetElementPtrInst>(Ptr);
170  }
171  while (GEP != nullptr);
172 
173  return nullptr;
174 }

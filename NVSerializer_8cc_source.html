95 NVSerializer::Serialize(StringRef Name, StringRef Descrip,
96  ArrayRef<Value*> Values, IRBuilder<>& B) {
97 
98  Value *NVList = NV->Create(B);
99  NV->Add(NVList, "name", Name, B);
100  NV->Add(NVList, "description", Descrip, B);
101 
102  Value *SubList = NV->Create(B);
103  for (Value *V : Values) {
104  NV->Add(SubList, V->getName(), V, B);
105  }
106 
107  NV->Add(NVList, "values", SubList, B);
108  NV->Destroy(SubList, B);
109 
110  if (NVDebug) {
111  NV->Dump(NVList, ConstantInt::get(NV->Int, 2 /* stderr */), B);
112  }
113 
114  Value *SizePtr = B.CreateAlloca(SizeT);
115  Value *Buffer = NV->Pack(NVList, SizePtr, B);
116 
117  // Destroy the nvlist_t instances: all we is now in the packed buffer.
118  NV->Destroy(NVList, B);
119 
120  return { Buffer, B.CreateLoad(SizePtr) };
121 }

177 void LibNV::Add(Value *List, StringRef Name, Value *V, IRBuilder<>& B) {
178  Type *T = V->getType();
179  Constant *F = nullptr;
180 
181  if (T->isFloatTy() or T->isDoubleTy()) {
182  llvm::errs() << "WARNING: libnv doesn't support floating-point numbers\n";
183 
184  } else if (auto *IT = dyn_cast<IntegerType>(T)) {
185  if (IT->getBitWidth() == 1) {
186  IntegerType *Bool = IntegerType::get(Ctx, 1);
187  F = Fn("nvlist_add_bool", Void, { NVListPtr, BytePtr, Bool });
188 
189  } else {
190  F = Fn("nvlist_add_number", Void, { NVListPtr, BytePtr, SizeT });
191 
192  if (T != SizeT) {
193  V = B.CreateSExt(V, SizeT);
194  }
195  }
196 
197  } else if (T->isPointerTy()) {
198  if (T == NVListPtr) {
199  F = Fn("nvlist_add_nvlist", Void, { NVListPtr, BytePtr, NVListPtr });
200 
201  } else {
202  F = Fn("nvlist_add_number", Void, { NVListPtr, BytePtr, SizeT });
203  V = B.CreatePointerCast(V, SizeT);
204  }
205 
206  } else {
207  raw_ostream& err = llvm::errs();
208  err << "WARNING: NVSerializer doesn't support ";
209  T->print(err, true);
210  err << " (yet)\n";
211  }
212 
213  if (F == nullptr)
214  return;
215 
216  Value *NameVal = B.CreateGlobalStringPtr(Name);
217 
218  B.CreateCall(F, { List, NameVal, V });
219 }
